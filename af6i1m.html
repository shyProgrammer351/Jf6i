문제

2km까지 기본료가 3800원이고 132m 당 100원이 부과된다고 가정할 때, 거리(m)를 입력하면 택시비를 계산하는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 
택시의 이동거리(m)를 입력하세요: 2000
2000m 이동시 택시요금은 3800원입니다

프로그램 코드

	#include <stdio.h>
	
1	int getDistance()																			
	{
		int distance = 0;																		
		printf( "택시의 이동거리(m)를 입력하세요: " );
		scanf( "%d", &distance ); 		
	
2		return distance;							
	}
	
3	int calculateTaxiFare( int distance )									
	{
		int taxiFare = 0;																		
		
		distance = ( distance < 2000 ) ? 2000 : distance;		
	
		taxiFare = 3800 + ( distance - 2000 + 131 ) / 132 * 100;			
	
4		return taxiFare;							
	}
	
5	int main()									
	{
		int distance = 0, taxiFare = 0;						
	
6		distance = 								
7			   getDistance();		 
	
8		taxiFare = 								
9			   calculateTaxiFare( distance );	 
	
10		printf( "%dm 이동시 택시요금은 %d원입니다 ", distance, taxiFare );
		
11		return 0;								
	}
