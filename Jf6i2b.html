import javax.swing.;

package threeCupGame; 
import java.awt.*; 
import java.awt.event.*; 
import javax.swing.*; 
 
public  ThreeCupGamePanel    	 
{ 
	protected int player; 
	protected int ball; 
	private [] button;	 
	private  timer; 
	protected  label;	 
	private [] cup;	 
	private [] img;	 
	private final int BASE = 0; 
	private final int BALL = 1; 
	private final int MISS = 2; 
 
	public ThreeCupGamePanel()  
	{ 
		img = new [3]; 
		final String imgPath = "./Common/ThreeCupGame/image/"; 
		final String PathImgCup  = imgPath + "cup.jpg"; 
		final String PathImgBall = imgPath + "cup_ball.jpg"; 
		final String PathImgFail = imgPath + "cup_fail.jpg";		 
		img[ BASE ] = new ( new ( PathImgCup  ).getImage().getScaledInstance( 200, 250, Image.SCALE_SMOOTH ) ); 
		img[ BALL ] = new ( new ( PathImgBall ).getImage().getScaledInstance( 200, 250, Image.SCALE_SMOOTH ) ); 
		img[ MISS ] = new ( new ( PathImgFail ).getImage().getScaledInstance( 200, 250, Image.SCALE_SMOOTH ) ); 
 
		cup = new [3]; 
		cup[0] = new ( img[ BASE ] ); 
		cup[1] = new ( img[ BASE ] ); 
		cup[2] = new ( img[ BASE ] ); 
		this.add( cup[0] ); 
		this.add( cup[1] ); 
		this.add( cup[2] ); 
 
		button = new [3]; 
		button[0] = new ( "                          1                           " ); 
		button[0].( this );  
 
		button[1] = new ( "                          2                           " ); 
		button[1].( this );  
 
		button[2] = new ( "                          3                           " ); 
		button[2].( this );  
 
		label = new ( "구슬을 숨긴 컵을 선택하세요" ); 
 
		this.add( button[0] ); 
		this.add( button[1] ); 
		this.add( button[2] ); 
		this.add( label ); 
 
		this.setBackground(Color.cyan); 
		this.setPreferredSize(new Dimension(630, 320));	 
 
		this.timer = new ( 1000, new TimerListener() ); 
	} 
	 
	protected int selectCup( ActionEvent event ) 
	{ 
		if( event.getSource() == button[0] )  
			return 0; 
		else if ( event.getSource() == button[1] )  
			return 1; 
		else 
			return 2; 
	} 
	 
	protected void showGameResult( int player, int ball )  
	{ 
		cup[ ball ].( img[ BALL ] );  
		if ( player != ball ) 
			cup[ player ].( img[ MISS ] ); 	 
	}  
	 
	public void ( ActionEvent event )   
	{ 
		this.ball = (int)( Math.random() * 3 ); 
		this.player =  
		               this.selectCup( event ); 
		showGameResult( this.player, this.ball ); 
		 
		timer.();		 
		button[0].( false ); 
		button[1].( false ); 
		button[2].( false ); 
	}	 
	 
	private  TimerListener  ActionListener 
	{ 
		public void ( ActionEvent event )  
		{ 
			timer.();			 
			cup[0].( img[ BASE ] );  
			cup[1].( img[ BASE ] );  
			cup[2].( img[ BASE ] ); 	 
			button[0].( true ); 
			button[1].( true ); 
			button[2].( true ); 
		} 
	} 
}
