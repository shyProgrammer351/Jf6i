			else if ( robotX >= ( WIDTH - IMAGE_SIZE ) )

				robotX = WIDTH - IMAGE_SIZE;


			if ( robotY <= 0 )

				robotY = 0;

				robotY = ( HEIGHT - IMAGE_SIZE );
