package rockPaperScissors; 
import java.awt.*; 
import java.awt.event.*; 
import javax.swing.*; 
 
public  RockPaperScissorsPanel       
{ 
	protected RockPaperScissors game; 
	private    timer; 
	 
	protected int[] playerA; 
	protected int[] playerB; 
 
	protected  imgPlayerA; 
	protected  imgPlayerB; 
 
	protected  strPlayers; 
 
	protected [] button;	 
	protected [][] image; 
	 
	protected final int BASE = 0; 
	protected final int WIN = 1; 
	protected final int QUESTIONMARK = 3; 
 
	public RockPaperScissorsPanel()  
	{ 
		this.game = new RockPaperScissors(); 
		this.timer = new ( 1000, new TimerListener() ); 
		 
		this.playerA = new int[2]; 
		this.playerB = new int[2]; 
		this.playerA[0] = this.playerB[0] = game.SCISSORS; 
		this.playerA[1] = this.playerB[1] = BASE; 
		 final String imgPath = "./Common/rockPaperScissors/image/"; 
		image = new [4][2]; 
		final String pathImgScissors = imgPath + "scissors.gif"; 
		final String pathImgRock = imgPath + "rock.gif"; 
		final String pathImgPaper = imgPath + "paper.gif"; 
		final String pathImgScissorsWin = imgPath + "scissors_win.gif"; 
		final String pathImgRockWin = imgPath + "rock_win.gif"; 
		final String pathImgPaperWin = imgPath + "paper_win.gif"; 
		final String pathImgQuestionmark = imgPath + "questionmark.gif"; 
		image[QUESTIONMARK][BASE] = new ( new ( pathImgQuestionmark ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[QUESTIONMARK][WIN] = new ( new ( pathImgQuestionmark ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[game.SCISSORS][BASE] = new ( new ( pathImgScissors ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[game.SCISSORS][WIN] = new ( new ( pathImgScissorsWin ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[game.ROCK][BASE] = new ( new ( pathImgRock ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[game.ROCK][WIN] = new ( new ( pathImgRockWin ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[game.PAPER][BASE] = new ( new ( pathImgPaper ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
		image[game.PAPER][WIN] = new ( new ( pathImgPaperWin ).getImage().getScaledInstance( 200, 200, Image.SCALE_SMOOTH ) ); 
			 
		this.strPlayers = new ( "<html>[너]<br>[나]</body></html>", .CENTER );		 
		this.imgPlayerA = new ( image[QUESTIONMARK][BASE] ); 
		this.imgPlayerB = new ( image[QUESTIONMARK][BASE] ); 
		 
		this.add( this.imgPlayerB ); 
		this.add( this.strPlayers ); 
		this.add( this.imgPlayerA ); 
 
		button = new [3]; 
		button[game.SCISSORS] = new ( new ( image[game.SCISSORS][BASE].getImage().getScaledInstance( 30, 30, Image.SCALE_SMOOTH ) ) ); 
		button[game.SCISSORS].( this );  
 
		button[game.ROCK] = new ( new ( image[game.ROCK][BASE].getImage().getScaledInstance( 30, 30, Image.SCALE_SMOOTH ) ) ); 
		button[game.ROCK].( this );  
 
		button[game.PAPER] = new ( new ( image[game.PAPER][BASE].getImage().getScaledInstance( 30, 30, Image.SCALE_SMOOTH ) ) ); 
		button[game.PAPER].( this );  
		this.add( new ("  가위 바위 보를 선택하세요  ") ); 
		this.add( button[game.SCISSORS] ); 
		this.add( button[game.ROCK] ); 
		this.add( button[game.PAPER] ); 
 
		this.setBackground(Color.cyan); 
		this.setPreferredSize(new Dimension(220, 530));		 
	} 
 
	protected int selectRockPaperScissors( ActionEvent event ) 
	{ 
		if( event.getSource() == button[game.SCISSORS] )  
			return game.SCISSORS; 
		else if ( event.getSource() == button[game.ROCK] )  
			return game.ROCK; 
		else 
			return game.PAPER; 
	} 
	 
	protected void setGameResult()  
	{ 
		String result =  
		                game.judge( playerA[0], playerB[0] ); 
		if ( result.contains( "참여자가 이겼습니다" ) ) 
		{ 			 
			playerA[1] = WIN; 
			playerB[1] = BASE; 
		}  
		else if ( result.contains( "상대편이 이겼습니다" ) ) 
		{ 			 
			playerA[1] = BASE; 
			playerB[1] = WIN; 
		}  
		else 
		{ 
			playerA[1] = BASE; 
			playerB[1] = BASE;	 
		} 
	}  
		 
	public void ( ActionEvent event )   
	{ 
		playerA[0] =  
		             selectRockPaperScissors( event ); 
		playerB[0] =  
		             game.selectRockPaperScissors(); 
 
		setGameResult(); 
		imgPlayerA.( image[ playerA[0] ][ playerA[1] ] ); 
		imgPlayerB.( image[ playerB[0] ][ playerB[1] ] );	 
		 
		button[0].( false ); 
		button[1].( false ); 
		button[2].( false ); 
		timer.();		 
	} 
	 
	private  TimerListener    
	{ 
		public void ( ActionEvent event )  
		{ 
			timer.();		 
			imgPlayerB.( image[QUESTIONMARK][BASE] ); 
			button[0].( true ); 
			button[1].( true ); 
			button[2].( true ); 
		} 
	} 
}	 
package rockPaperScissors; 
import java.util.Scanner; 
 
public  RockPaperScissors 
{ 
	public final int SCISSORS = 0; 
	public final int ROCK = 1; 
	public final int PAPER = 2; 
 
	public String play( Scanner scan ) 
	{	 
		int playerA = selectRockPaperScissors( scan ); 
		int playerB = selectRockPaperScissors(); 
		 
		String result = ""; 
		result += "참여자는 " + getString( playerA ) +"를 냈습니다.\n"; 
		result += "상대편은 " + getString( playerB ) +"를 냈습니다.\n"; 
		result += "판정결과는 " + judge( playerA, playerB ) + "\n"; 
		return result; 
	} 
	 
	public String getString( int player ) 
	{ 
		if ( player == SCISSORS ) 
			return "가위"; 
		else if ( player == ROCK ) 
			return "바위"; 
		else 
			return "보";  
	} 
	 
	public int getInteger( String player ) 
	{ 
		if ( player.equals( "가위" ) ) 
			return SCISSORS; 
		else if ( player.equals( "바위" ) ) 
			return ROCK; 
		else 
			return PAPER;  
	} 
	 
	protected int selectRockPaperScissors( Scanner scan ) 
	{ 
		String player; 
		 
		do 
		{ 
			System.out.print( "가위, 바위, 보 중 하나를 입력하세요 : "); 
			player = scan.next(); 
			 
		} while( !player.equals( "가위" ) && !player.equals( "바위" ) && !player.equals( "보" ) ); 
			 
		return getInteger( player ); 
	} 
 
	protected int selectRockPaperScissors() 
	{ 
		return (int)(Math.random()*3);		 
	} 
	 
	protected String judge( int playerA, int playerB ) 
	{ 
		if ( playerA == playerB ) 
		{ 
			return "상대편과 참여자가 비겼습니다"; 
		} 
		else if ( ( ( playerA == SCISSORS ) && ( playerB == PAPER ) ) 
				|| ( ( playerA == ROCK ) && ( playerB == SCISSORS ) ) 
				|| ( ( playerA == PAPER ) && ( playerB == ROCK ) ) ) 
		{ 
			return "참여자가 이겼습니다"; 
		} 
		else 
		{ 
			return "상대편이 이겼습니다"; 
		}	 
	} 
}
